001   "use client";  

   002   import LiveDiv from "@/components/liveDiv";  
   003   import { ChevronDownIcon } from "@heroicons/react/24/outline";  

   004   import Lightbox from "yet-another-react-lightbox";  
   005   import "yet-another-react-lightbox/styles.css";  

   006   import Counter from "yet-another-react-lightbox/plugins/counter";  
   007   import "yet-another-react-lightbox/plugins/counter.css";  

   008   import Thumbnails from "yet-another-react-lightbox/plugins/thumbnails";  
   009   import "yet-another-react-lightbox/plugins/thumbnails.css";  

   010   import Fullscreen from "yet-another-react-lightbox/plugins/fullscreen";  

   011   import Zoom from "yet-another-react-lightbox/plugins/zoom";  

   012   import NextJsImage from "@/components/NextJsImage";  
   013   import NextJsThumbnail from "@/components/NextJsThumbnail";  

   014   import Image from "next/image";  
   015   import Heading from "@/components/heading";  
   016   import { useRef, useState, useEffect } from "react";  
   017   import { useScroll } from "@react-hooks-library/core";  
   018   import { notFound } from "next/navigation";  
   019   import Footer from "@/components/footer";  

   020   interface ImageProps {  
   021     src: string;  
   022     alt: string;  
   023   }  
   024   interface InfoProps {  
   025     name: string;  
   026     description?: string;  
   027   }  

   028   const Object = ({ params }: { params: { id: number } }) => {  
   029     if (isNaN(params.id) || params.id < 1 || params.id > 12) {  
   030       notFound();  
   031     }  

   032     const [imageArray, setImageArray] = useState<ImageProps[]>([  
   033       { src: "", alt: "" },  
   034     ]);  
   035     const [info, setInfo] = useState<InfoProps>({  
   036       name: "Layih…ô",  
   037       description: "",  
   038     });  

   039     const [windowDimensions, setWindowDimensions] = useState<{  
   040       width: number;  
   041       height: number;  
   042     }>({  
   043       width: 0,  
   044       height: 0,  
   045     });  

   046     useEffect(() => {  
   047       if (typeof window !== "undefined") {  
   048         const width = window.innerWidth;  
   049         const height = window.innerHeight;  
   050         if (width < height) {  
   051           setWindowDimensions({ width: 1080, height: 1920 });  
   052         } else {  
   053           setWindowDimensions({ width: 1920, height: 1080 });  
   054         }  
   055       }  
   056     }, [typeof window !== "undefined" && window.innerWidth]);  

   057     useEffect(() => {  
   058       (async () => {  
   059         const res = await import(`../${params.id}.json`);  
   060         setImageArray(res.images);  
   061         setInfo(res.info);  
   062       })();  
   063     }, []);  

   064     const imageContainer = useRef(null);  
   065     const [imageScroll, setImageScroll] = useState({ x: 0, y: 0 });  
   066     useScroll(imageContainer, ({ scrollX, scrollY }) => {  
   067       setImageScroll({ x: scrollX, y: scrollY });  
   068     });  

   069     const [lightboxOpen, setLightboxOpen] = useState(false);  
   070     const [index, setIndex] = useState(-1);  
   071     const thumbnailsRef = useRef(null);  

   072     const imageSizes = [16, 32, 48, 64, 96, 128, 256, 384];  
   073     const deviceSizes = [640, 750, 828, 1080, 1200, 1920, 2048, 3840];  

   074     function nextImageUrl(src: string, size: number) {  
   075       return `/_next/image?url=${encodeURIComponent(`/objects/${params.id}/images/${src}`)}&w=${size}&q=70`;  
   076     }  

   077     interface Slide {  
   078       src: string;  
   079       width?: number;  
   080       height?: number;  
   081       alt: string;  
   082     }  
   083     const slides = imageArray.map(  
   084       ({  
   085         src,  
   086         width = windowDimensions.width,  
   087         height = windowDimensions.height,  
   088         alt,  
   089       }: Slide) => ({  
   090         width,  
   091         height,  
   092         src: nextImageUrl(src, width),  
   093         srcSet: imageSizes  
   094           .concat(...deviceSizes)  
   095           .filter((size) => size <= width)  
   096           .map((size) => ({  
   097             src: nextImageUrl(src, size),  
   098             width: size,  
   099             height: Math.round((height / width) * size),  
   100           })),  
   101         alt: alt,  
   102       }),  
   103     );  

   104     return (  
   105       <main className="bg-grayA">  
   106         <div className="px-8 lg:pt-24 pt-12 lg:pb-12">  
   107           <div className="w-full lg:mb-8 mb-2 flex items-end">  
   108             <Heading  
   109               animate={{ from: -10, to: 0, dir: "x" }}  
   110               variant="h3"  
   111               text={info.name}  
   112             />  
   113             <LiveDiv animate={{ from: 10, to: 0, dir: "x", delay: 0.1 }}>  
   114               <p className="lg:text-xl text-sm ml-auto w-fit text-neutral-400">  
   115                 {info.description}  
   116               </p>  
   117             </LiveDiv>  
   118           </div>  
   119           <Lightbox  
   120             plugins={[Counter, Fullscreen, Thumbnails, Zoom]}  
   121             thumbnails={{  
   122               ref: thumbnailsRef,  
   123               position: "bottom",  
   124               width: 200,  
   125               height: 150,  
   126               padding: 2,  
   127               border: 0,  
   128               gap: 0,  
   129               imageFit: "cover",  
   130               vignette: false,  
   131             }}  
   132             zoom={{  
   133               scrollToZoom: true,  
   134             }}  
   135             on={{  
   136               entering: () => {  
   137                 if (window.innerWidth < 640) {  
   138                   (thumbnailsRef.current as any)?.hide();  
   139                 } else {  
   140                   (thumbnailsRef.current as any)?.show();  
   141                 }  
   142               },  
   143             }}  
   144             open={lightboxOpen}  
   145             index={index}  
   146             close={() => setLightboxOpen(false)}  
   147             slides={slides}  
   148             render={{  
   149               slide: NextJsImage,  
   150               thumbnail: NextJsThumbnail as any,  
   151             }}  
   152           />  
   153           <div className="w-full flex lg:flex-row flex-col">  
   154             <div className="lg:w-1/2 w-full aspect-[4/3]">  
   155               <Image  
   156                 onClick={() => {  
   157                   setIndex(0);  
   158                   setLightboxOpen(true);  
   159                 }}  
   160                 className="object-cover w-full h-full"  
   161                 quality={70}  
   162                 src={`/objects/${params.id}/images/hero.jpg`}  
   163                 width={1080}  
   164                 height={0}  
   165                 priority  
   166                 alt={imageArray[0].alt}  
   167               />  
   168             </div>  
   169             <div  
   170               ref={imageContainer}  
   171               className="relative lg:w-1/2 w-full lg:aspect-[4/3] lg:h-auto flex flex-col  
   172                 lg:flex-row flex-wrap overflow-scroll gap-2 lg:pl-2 pt-2 lg:pt-0 h-48"  
   173             >  
   174               {imageArray  
   175                 .filter((_, index) => index !== 0)  
   176                 .map((image, index) => (  
   177                   <div  
   178                     key={index}  
   179                     className="lg:w-[calc(33.3333%-8px)] lg:aspect-auto lg:h-1/2  
   180                         lg:overflow-auto overflow-y-hidden w-1/2 aspect-[4/3]"  
   181                   >  
   182                     <Image  
   183                       onClick={() => {  
   184                         setIndex(index + 1);  
   185                         setLightboxOpen(true);  
   186                       }}  
   187                       priority  
   188                       className="object-cover w-full h-full"  
   189                       width={720}  
   190                       height={0}  
   191                       src={`/objects/${params.id}/images/${image.src}`}  
   192                       alt={image.alt}  
   193                       quality={50}  
   194                     />  
   195                   </div>  
   196                 ))}  
   197               <span  
   198                 className="absolute sm:flex hidden items-center justify-center  
   199                   bottom-0 left-1/2 -translate-x-1/2 -translate-y-1/2"  
   200               >  
   201                 {imageScroll.y === 0 && imageArray.length > 6 && (  
   202                   <ChevronDownIcon className="opacity-0 lg:w-12 text-zinc-100/60 animate-down" />  
   203                 )}  
   204               </span>  
   205             </div>  
   206           </div>  
   207         </div>  
   208         <Footer />  
   209       </main>  
   210     );  
   211   };  
   212   export default Object;
